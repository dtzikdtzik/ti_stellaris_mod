trait_brilliant = {
	cost = 2
	category = normal
	species_possible_add = {
		can_add_genetic_traits = yes
	}
	species_possible_remove = {
		OR = {
			can_remove_beneficial_genetic_traits = yes
			can_remove_vocational_genetic_traits = yes
		}
	}
	species_possible_merge_remove = {
		always = yes
	}
	opposites = { "trait_nerve_stapled" "trait_erudite" "trait_enigmatic_intelligence_poor" }
	allowed_archetypes = { BIOLOGICAL LITHOID }
	localized_tags = {
		SPECIES_TRAIT_VOCATIONAL
	}
	tags = { organic positive pop_output research }
	modifier = {
		researcher_jobs_bonus_workforce_mult = 0.05
		planet_researchers_upkeep_mult = -0.01
		
	}
	slave_cost = {
		energy = 500
	}
	assembly_score = {
		modifier = {
			add = 1.5
			from = { has_research_designation = yes }
		}
		modifier = {
			add = 1.5
			from = { has_building = building_order_keep }
		}
	}
}

trait_zeal = {
	cost = 2
	species_possible_add = {
		can_add_genetic_traits = yes
	}
	species_possible_remove = {
		OR = {
			can_remove_beneficial_genetic_traits = yes
		}
	}
	allowed_archetypes = { BIOLOGICAL LITHOID }
	tags = { organic positive pop_output unity }
	icon = "gfx/interface/icons/traits/trait_existential_iteroparity.dds"
	custom_tooltip = TRAIT_ZEAL_EFFECT
	triggered_pop_group_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					is_at_war = yes
					has_crisis = yes
				}
			}
		}
		pop_soldier_bonus_workforce_mult = 0.25
		pop_high_priest_bonus_workforce_mult = 0.25
		pop_bureaucrat_bonus_workforce_mult = 0.25
		pop_happiness_negative_slave = 20
	}
	slave_cost = {
		energy = 1000
	}
}
